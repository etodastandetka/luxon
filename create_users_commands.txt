# КОМАНДЫ ДЛЯ СОЗДАНИЯ ПОЛЬЗОВАТЕЛЕЙ В DJANGO SHELL

# Запустить Django shell:
cd django_admin
python manage.py shell

# В shell выполнить:
from django.contrib.auth.models import User

# Создать суперпользователя
admin = User.objects.create_superuser('admin', 'admin@luxon.com', 'admin123')
print("Создан admin / admin123")

# Создать менеджера
manager = User.objects.create_user('manager', 'manager@luxon.com', 'manager123', is_staff=True)
print("Создан manager / manager123")

# Создать тестового пользователя
test = User.objects.create_user('test', 'test@luxon.com', 'test123', is_staff=True)
print("Создан test / test123")

# Выйти из shell
exit()
